  , Q = require('q')

var runner = exports = module.exports = {};
 
var promiseExec = Q.denodeify(exec);
 
var getDiskImageInfo = function () {
    // Attempt to find the OS version of the iDevice, e.g. 7.1
    var versionInfo = promiseExec("ideviceinfo -s -k ProductVersion").then(function (args) {
        var stdout = args[0];
        return stdout.trim().substring(0, 3); // Versions for DeveloperDiskImage seem to be X.Y, while some device versions are X.Y.Z
        // NOTE: This will almost certainly be wrong in the next few years, once we hit version 10.0 
    }, function (err) {
        throw new Error("Can't get device info");
    });
    
    // Attempt to find the path where developer resources exist.
    var pathInfo = promiseExec("xcrun -sdk iphoneos --show-sdk-platform-path").then(function (args) {
        var sdkpath = args[0].trim();
        return sdkpath;
    });
    
    // Attempt to find the developer disk image for the appropriate 
    return Q.all([versionInfo, pathInfo]).spread(function (version, sdkpath) {
        var spaceSafeName = util.quotesAroundIfNecessary(sdkpath);
        return promiseExec('find ' + spaceSafeName + ' -path "*' + version + '*" -name "DeveloperDiskImage.dmg"');
    });
}
 
var mountDeveloperImage = function () {
    return getDiskImageInfo().then(function (args) {
        var findStdOut = args[0];
        var candidate = findStdOut.split("\n")[0];
        var dmgPath = candidate.trim();
        if (dmgPath == "") {
            throw new Error("Unable to find DeveloperDiskImage.dmg");
        }
        var quoteSafeDmgPath = util.quotesAroundIfNecessary(dmgPath);
        var deferred = Q.defer();
        exec('ideviceimagemounter ' + quoteSafeDmgPath, function (err, stdout, stderr) {
            if (err) {
                if (stdout.indexOf("Error:") > -1) {
                    deferred.resolve('Failed; may already be mounted?');
                } else if (stdout.indexOf('No device found, is it plugged in?') > -1) {
                    deferred.reject(new Error("No Device Attached"));
                }
            } else {
                deferred.resolve(stdout);
            }
        });
        return deferred.promise;
    });
}

runner.shutdown = function () {
    if (proxyInstance) {
        console.info("Shutting down native debug proxy...");
        proxyInstance.kill('SIGHUP');
        proxyInstance = null;
    }
    if (webProxyInstance) {
        console.info("Shutting down web debug proxy...");
        webProxyInstance.kill();
        webProxyInstance = null;
    }
}
 
// Attempt to start the debug server proxy listening on a given port. Returns the executable, so it can be stopped later if necessary.
runner.startDebugProxy = function (portNumber) {
    if (proxyInstance != null) {
        proxyInstance.kill('SIGHUP'); // idevicedebugserver does not exit from SIGTERM
        proxyInstance = null;
    }
    return mountDeveloperImage().then(function () {
        proxyInstance = exec('idevicedebugserverproxy ' + portNumber);
        return proxyInstance;
    });
}
 
runner.startiOSWebkitDebugProxy = function (devicePortNumber, portRangeMin, portRangeMax) {
    if (webProxyInstance != null) {
        webProxyInstance.kill(); // sigterm works fine and allows for a graceful exit.
        webProxyInstance = null;
    }
    var commandString = 'ios_webkit_debug_proxy -c null:' + devicePortNumber + ",:" + portRangeMin + "-" + portRangeMax;
    console.info("Running command '"+commandString+"'");
    webProxyInstance = exec(commandString);
    var deferred = Q.defer();
    setTimeout(function () {
        if (!webProxyInstance) {
            deferred.reject('WebkitProxyFailed');
        }
        if (webProxyInstance.exitCode) {
            deferred.reject('WebkitProxyFailed');
        }
        deferred.resolve(webProxyInstance);
    }, 250);
    return deferred.promise;
}
 
runner.encodePath = function (packagePath) {
    // Encode the path by converting each character value to hex
    var encodedPath = "";
    for (var i = 0; i < packagePath.length; ++i) {
        var c = packagePath[i];
        encodedPath += charToHex(c);
    }
    return encodedPath;
}
 
runner.startAppViaDebugger = function (portNumber) {
    return function (packagePath) {
        var encodedPath = runner.encodePath(packagePath);
        
        // We need to send 3 messages to the proxy, waiting for responses between each message:
        
        // A(length of encoded path),0,(encoded path)
        // Hc0
        // c
        
        // We expect a '+' for each message sent, followed by a $OK#9a to indicate that everything has worked.
        
        // For more info, see http://www.opensource.apple.com/source/lldb/lldb-167.2/docs/lldb-gdb-remote.txt
        
        var socket = new net.Socket();
        socket.initState = 0;
        socket.endStatus = null;
        socket.endSignal = null;
        
        var deferred1 = Q.defer();
        var deferred2 = Q.defer();
        var deferred3 = Q.defer();
        
        socket.on("data", function (data) {
            data = data.toString();
            while (data[0] == '+') { data = data.substring(1); }
            // Acknowledge any packets sent our way
            if (data[0] == '$') {
                socket.write('+');
                if (data[1] == 'W') {
                    // The app process has exited, with hex status given by data[2-3] 
                    var status = parseInt(data.substring(2, 4), 16)
                    socket.endStatus = status;
                    socket.end();
                } else if (data[1] == 'X') {
                    // The app rocess exited because of signal given by data[2-3]
                    var signal = parseInt(data.substring(2, 4), 16)
                    socket.endSignal = signal;
                    socket.end();
                } else if (data.substring(1, 3) == 'OK') {
                    // last command was received OK;
                    if (socket.initState == 1) {
                        deferred1.resolve(socket);
                    } else if (socket.initState == 2) {
                        deferred2.resolve(socket);
                    }
                } else if (data[1] == 'O') {
                    // STDOUT was written to, and the rest of the input until reaching a '#' is a hex-encoded string of that output
                    if (socket.initState == 3) {
                        console.info("App started!");
                        deferred3.resolve(socket);
                        socket.initState++;
                    }
                } else if (data[1] == 'E') {
                    // An error has occurred, with error code given by data[2-3]
                    deferred1.reject("Launch failed: " + parseInt(data.substring(2, 4), 16));
                    deferred2.reject("Launch failed: " + parseInt(data.substring(2, 4), 16));
                    deferred3.reject("Launch failed: " + parseInt(data.substring(2, 4), 16));
                }
                
            }
        });
        
        socket.on("end", function () {
            deferred1.reject("Unable to launch app");
            deferred2.reject("Unable to launch app");
            deferred3.reject("Unable to launch app");
        });
 
        socket.connect(portNumber, "localhost", function () {
            // set argument 0 to the (encoded) path of the app
            console.info("Starting app...");
            var cmd = makeGdbCommand("A" + encodedPath.length + ",0," + encodedPath);
            socket.initState++;
            socket.write(cmd);
            setTimeout(function () {
                if (socket.initState == 1) {
                    deferred1.reject("Timeout: Is the device locked?");
                    deferred2.reject("Timeout: Is the device locked?");
                    deferred3.reject("Timeout: Is the device locked?");
                    socket.end();
                }
            }, 5000);
        });
        
        return deferred1.promise.then(function (sock) {
            // Set the step and continue thread to any thread
            var cmd = makeGdbCommand("Hc0");
            sock.initState++;
            sock.write(cmd);
            setTimeout(function () {
                if (socket.initState == 2) {
                    deferred2.reject("Timeout: Is the device locked?");
                    deferred3.reject("Timeout: Is the device locked?");
                    socket.end();
                }
            }, 5000);
            return deferred2.promise;
        }).then(function (sock) {
            //Continue execution; actually start the app running.
            var cmd = makeGdbCommand("c");
            sock.initState++;
            sock.write(cmd);
            setTimeout(function () {
                if (socket.initState == 3) {
                    deferred3.reject("Timeout: Is the device locked?");
                    socket.end();
                }
            }, 5000);
            return deferred3.promise;
        });
    }
}
 
// Attempt to start the app on the device, using the debug server proxy on a given port.
// Returns a socket speaking remote gdb protocol with the debug server proxy.
runner.startApp = function (packageId, portNumber) {
    // When a user has many apps installedl on their device, the response from ideviceinstaller may be large (500k or more)
    // This exceeds the maximum stdout size that exec allows, so we redirect to a temp file.
    return promiseExec('ideviceinstaller -l -o xml > /tmp/$$.ideviceinstaller && echo /tmp/$$.ideviceinstaller').then(function (args) {
        // First find the path of the app on the device
        var filename = args[0].trim();
        if (! /^\/tmp\/[0-9]+\.ideviceinstaller$/.test(filename)) {
            throw new Error("Output file not as expected");
        }
        var list = pl.parseFileSync(filename);
        fs.unlink(filename);
        for (var i = 0; i < list.length; ++i) {
            if (list[i].CFBundleIdentifier == packageId) {
                return list[i].Path;
            }
        }
        throw new Error("Package not installed?");
    }).then(runner.startAppViaDebugger(portNumber));
}
 
function charToHex(char) {
    var conversionTable = '0123456789ABCDEF';
    var charCode = char.charCodeAt(0);
    return conversionTable[(charCode & 0xF0) >> 4] + conversionTable[charCode & 0x0F];
}
 
function makeGdbCommand(command) {
    var commandString = "$" + command + "#";
    var stringSum = 0;
    for (var i = 0; i < command.length; i++) {
        stringSum += command.charCodeAt(i);
    }
    stringSum = stringSum & 0xFF;
    var checksum = stringSum.toString(16).toUpperCase();
    if (checksum.length < 2) {
        checksum = '0' + checksum;
    }
    commandString += checksum;
    return commandString;
}
